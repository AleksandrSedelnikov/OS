ping -c 1 $1 | grep "100% packet loss" # пинг введённого ip хоста
if [ $? -eq 0 ] # если возврат команды выше равен 0
then # то
echo "хост недоступен" 
else # иначе
echo "символьное имя доменное имя"
result=`host $1 | awk '{print $5}'` # из команды host вырезается 5 столбец с символьным названием домена
if [ "$result" != "3(NXDOMAIN)" ] # если название домена не равно 3(...)
then # то
echo $result
else # иначе
echo "символьного имени нет."
fi
fi
