if [ $# -eq 0 ] || [ $# -eq 1 ] || [ $# -gt 10 ] # проверка что введено не 0, 1, 10 параметров
then
echo -e "Нарушен синтаксис.\nРазрешено вводить не более 10-ти параметров и не меньше 2-х.\nВведено: $# с учётом файла назначения" # если нарушен синтаксис
else # иначе
count=`echo $#` # переменная содержащая количество параметров
for ((i=1;i<=$count;i++)) # цикл по параметрам
do
if [ $i -eq 1 ] # если это первый параметр, то записать в переменную файл (это будет файл назначения)
then
file=`echo $1` # переменная файла назначения
else # иначе
echo -e "Параметр $1 добавлен в файл с именем $file."
echo $1 >> $file # добавляем параметр в файл назначения
fi
shift # смещаем строку до следующего параметра
done
fi
