if [ $# -eq 0 ] || [ $# -eq 1 ] || [ $# -gt 10 ] # проверка параметров
then # если провалено
echo -e "Нарушен синтаксис.\nРазрешено вводить не более 10-ти параметров и не меньше 2-х.\nВведено: $#"
else # успешно
count=`echo $#` # в переменную count записывается количество параметров
for ((i=1;i<=$count;i++)) # цикл по параметрам
do
if [ $i -eq 1 ] # если это первый параметр
then
file=`echo $1` # название файла записывается в переменную file
else # иначе
echo -e "Параметр $1 добавлен в файл с именем $file."
echo $1 >> $file
fi
shift # скип до следующего параметра
done
fi
