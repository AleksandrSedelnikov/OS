#RELEASE 1.0 (количество строчек: 44)


out3=0 # нулевая перемена для счёта количества строчек в файле
out1=0 # нулевая перемена для счёта количества строчек в файлах каталога
proverka=0 # нулевая перемена для счёта количества входов в остальные проверки
echo -n "Введите параметры через пробел: "; read parametr # вводится(ятся) параметр(ы), точнее: имя файла или каталога
count=$(echo $parametr | wc -w) # определение количества параметров
count_1=(${parametr}) # создание массива параметров разделенных пробелом
	for ((i=0;i<"$count";i++)) # цикл прогона от 0 до последний -1 параметра
	do
	count_2=`find ~ -name "${count_1[i]}" -type f 2>~/error | wc -l` # поиск введенного параметра по всем файлам системы
	count_3=`find ~ -name "${count_1[i]}" -type d 2>~/error | wc -l` # поиск введенного параметра по всем каталогам системы
	if [ "$count_2" == 1 ] || [ "$count_3" == 1 ] # проверка существует ли с таким именем файл или каталог
		then
			if [ "$count_2" == 1 ] && [ "$count_3" == 0 ] # проверка на то, что данный параметр файл
				then
					proverka1=`find ~ -name "${count_1[i]}" -type f 2>~/error` # поиск пути до данного файла
					out=`grep -c $ "${proverka1}"` # подсчёт количества строчек в данном файле
					out1=`expr $out + $out1` # обновление глобальной переменной подсчёта количества строчек в файле
					proverka=`expr $proverka + 1` # подсчёт количества вхождений в данную итерацию
				elif [ "$count_3" == 1 ] && [ "$count_2" == 0 ] # проверка на то, что данный параметр каталог
					then
						proverka2=`find ~ -name "${count_1[i]}" -type d 2>~/error` # поиск пути до данного каталога
						cd $proverka2 # смена директории прохождения в данный каталог
						for FILE in * # цикл прогона всех файлов в данном каталоге
						do
						if [ -f "$FILE" ]
						then  
							g=`grep -c $ $FILE` # подсчёт строчек в каждом файле данном каталоге
							out3=`expr $out3 + $g` # обновление глобальной переменной подсчёта строчек в файлах каталога
						fi
						done
						proverka=`expr $proverka + 1` # подсчёт количества вхождений в данную итерацию
			fi
	else
		echo "Error: ${count_1[i]} данный файл или каталог не прошёл проверку на единичность, cуществование или всё вместе." # ошибка если файл не прошёл первоначальную проверку на ЕП, СиФК или всё вместе.
	fi
	done
if [ "$proverka" != 0 ] # проверка на вхождение хоть в одну итерацию, а это значит что хоть один файл или каталог имеют единичный путь, существование или всё вместе
	then
		out_f=`expr $out1 + $out3` # подсчёт всех строчек в файле(ах), и файлах каталога(ов)
		echo "Количество строчек в файле(ах) или каталоге(ах): $out_f"
	else
		echo "Error code stopped script: 01" # ошибка при проверки СФиК
		echo "INFO: ошибка при выполнении проверки CФиК, точнее: Код 01 вызывается(был вызван) в случае срабатывания проверки(полученный ответ дал нулевое значение входов в остальные проверки) на существование файлов и каталогов."
fi
echo "Скрипт завершил работу." # завершение скрипта
